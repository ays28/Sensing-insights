function renderGraph(g) { graphData = g; if (!g || !g.nodes) return; const wrap = document.getElementById("graphWrap"); const svg = d3.select("#graphSvg"); const w = wrap.clientWidth; const h = wrap.clientHeight; svg.attr("viewBox", `0 0 ${w} ${h}`).selectAll("*").remove(); // Copy data const nodes = (g.nodes || []).map(d => ({...d})); const links = (g.edges || []).map(d => ({...d})); // Helpers const sevScale = d3.scaleLinear().domain([0, 100]).range([7, 18]); const nodeRadius = (d) => d.group === "market" ? 16 : sevScale(d.severity ?? 0); const nodeColor = (d) => { if (d.group === "market") return "#9bd0ff"; const s = d.severity ?? 0; if (s >= 80) return "#ff4d4d"; if (s >= 60) return "#ff9f1a"; if (s >= 40) return "#ffd166"; return "#6ee7ff"; }; const shortLabel = (d) => { const raw = (d.label || d.id || "").trim(); // keep market nodes readable if (d.group === "market") return raw.length > 20 ? raw.slice(0, 20) + "…" : raw; // events: short label only return raw.length > 26 ? raw.slice(0, 26) + "…" : raw; }; // Glow const defs = svg.append("defs"); const filter = defs.append("filter").attr("id", "glow"); filter.append("feGaussianBlur").attr("stdDeviation", "2.2").attr("result", "blur"); const merge = filter.append("feMerge"); merge.append("feMergeNode").attr("in", "blur"); merge.append("feMergeNode").attr("in", "SourceGraphic"); // Zoom const root = svg.append("g"); svg.call(d3.zoom().scaleExtent([0.6, 3]).on("zoom", (event) => { root.attr("transform", event.transform); })); // Links const link = root.append("g") .attr("stroke", "#22365c") .attr("stroke-opacity", 0.9) .selectAll("line") .data(links) .join("line") .attr("stroke-width", 1.2); // Nodes const node = root.append("g") .selectAll("circle") .data(nodes) .join("circle") .attr("r", d => nodeRadius(d)) .attr("fill", d => nodeColor(d)) .attr("stroke", "#0f1a33") .attr("stroke-width", 2) .style("filter", "url(#glow)") .call(d3.drag() .on("start", (event, d) => { if (!event.active) sim.alphaTarget(0.3).restart(); d.fx = d.x; d.fy = d.y; }) .on("drag", (event, d) => { d.fx = event.x; d.fy = event.y; }) .on("end", (event, d) => { if (!event.active) sim.alphaTarget(0); d.fx = null; d.fy = null; }) ); // Labels (short + non-overlapping-ish) const label = root.append("g") .selectAll("text") .data(nodes) .join("text") .attr("fill", "#e6e8ee") .attr("font-size", 11) .attr("paint-order", "stroke") .attr("stroke", "#0b1220") .attr("stroke-width", 4) .attr("stroke-linejoin", "round") .attr("pointer-events", "none") .text(d => shortLabel(d)); // Tooltip node.append("title").text(d => { const extra = d.group === "market" ? "Market node" : `${d.event_type || "event"} | sev ${d.severity ?? "?"}`; return `${d.label || d.id}\n${extra}`; }); // Click -> show inspector node.on("click", (event, d) => { const f = allFeatures.find(x => x.properties?.id === d.id); if (f) { selectedIndex = filteredFeatures.findIndex(x => x.properties?.id === d.id); renderSelected(f.properties); } else { document.getElementById("selected").innerHTML = `
${d.label || d.id}
Market node
`; } }); // Simulation (more spacing so it doesn't pile up) if (sim) sim.stop(); sim = d3.forceSimulation(nodes) .force("link", d3.forceLink(links).id(d => d.id).distance(140).strength(0.7)) .force("charge", d3.forceManyBody().strength(-520)) .force("center", d3.forceCenter(w/2, h/2)) .force("collision", d3.forceCollide().radius(d => nodeRadius(d) + 12)); sim.on("tick", () => { link .attr("x1", d => d.source.x).attr("y1", d => d.source.y) .attr("x2", d => d.target.x).attr("y2", d => d.target.y); node.attr("cx", d => d.x).attr("cy", d => d.y); label .attr("x", d => d.x + nodeRadius(d) + 8) .attr("y", d => d.y + 4); }); // Summary note const s = summarizeGraph(g); document.getElementById("graphStats").textContent = `${s.totalNodes} nodes • ${s.totalEdges} links`; document.getElementById("graphSummary").textContent = `Connected groups: ${s.clustersCount}. Isolated nodes: ${s.isolatedCount}.`; // (clusters rendering unchanged) }
